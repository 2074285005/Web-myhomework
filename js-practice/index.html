JavaScript 基础知识综合练习

1. 变量定义 & 变量类型练习

下面代码展示了 JS 中不同类型变量的定义（字符串、数字、布尔、数组、对象），打开浏览器控制台可查看结果：

// 1. 变量定义（使用 let/const 声明，避免 var 变量提升问题）
let userName = "前端学习小能手"; // 字符串类型
let userAge = 20; // 数字类型（Number）
let isStudent = true; // 布尔类型（Boolean）
let hobbies = ["JavaScript", "HTML", "CSS"]; // 数组类型（Array）
let userInfo = { // 对象类型（Object）
    major: "Web前端开发",
    school: "XX大学",
    grade: "2024级"
};

// 2. 打印变量类型（用 typeof 验证）
console.log("用户名：", userName, " | 类型：", typeof userName);
console.log("年龄：", userAge, " | 类型：", typeof userAge);
console.log("是否学生：", isStudent, " | 类型：", typeof isStudent);
console.log("爱好数组：", hobbies, " | 类型：", typeof hobbies);
console.log("用户信息对象：", userInfo, " | 类型：", typeof userInfo);

2. 表达式练习（算术/比较/逻辑运算）

覆盖常用表达式，直接复制到 script 标签中运行查看结果：

// 1. 算术表达式
let num1 = 10;
let num2 = 3;
let addResult = num1 + num2; // 加法
let subResult = num1 - num2; // 减法
let mulResult = num1 * num2; // 乘法
let divResult = num1 / num2; // 除法
let modResult = num1 % num2; // 取余

// 2. 比较表达式
let isEqual = num1 === num2; // 严格相等
let isGreater = num1 > num2; // 大于
let isLess = num1 < num2; // 小于

// 3. 逻辑表达式
let logicAnd = isGreater && isLess; // 逻辑与（且）
let logicOr = isGreater || isLess; // 逻辑或
let logicNot = !isEqual; // 逻辑非（取反）

// 打印表达式结果
console.log("10 + 3 =", addResult);
console.log("10 - 3 =", subResult);
console.log("10 * 3 =", mulResult);
console.log("10 / 3 =", divResult.toFixed(2)); // 保留2位小数
console.log("10 % 3 =", modResult);
console.log("10 === 3 ?", isEqual);
console.log("10 > 3 ?", isGreater);
console.log("逻辑与 &&：", logicAnd);
console.log("逻辑或 ||：", logicOr);

3. 流程控制练习（if-else / switch / for / while）

包含分支结构和循环结构，覆盖所有常用流程控制方式：

// 1. 分支流程 - if-else 语句
let score = 85; // 模拟考试分数
if (score >= 90) {
    console.log("成绩等级：A（优秀）");
} else if (score >= 80) {
    console.log("成绩等级：B（良好）");
} else if (score >= 60) {
    console.log("成绩等级：C（及格）");
} else {
    console.log("成绩等级：D（不及格）");
}

// 2. 分支流程 - switch 语句
let grade = "B";
switch (grade) {
    case "A":
        console.log("表现：优秀，奖励一次实践机会");
        break;
    case "B":
        console.log("表现：良好，继续保持");
        break;
    case "C":
        console.log("表现：及格，需要加强练习");
        break;
    default:
        console.log("表现：不及格，需补考");
}

// 3. 循环流程 - for 循环（遍历数组）
let courses = ["JavaScript", "HTML", "CSS", "Vue"];
console.log("--- for 循环遍历课程 ---");
for (let i = 0; i < courses.length; i++) {
    console.log("第", i+1, "门课：", courses[i]);
}

// 4. 循环流程 - while 循环（打印1-5）
console.log("--- while 循环打印1-5 ---");
let count = 1;
while (count <= 5) {
    console.log("计数：", count);
    count++;
}

4. 函数练习（声明/调用/参数/返回值）

封装常用功能，展示函数的定义、传参和返回值使用方式：

// 1. 普通函数声明（无返回值）
function greetUser(name) {
    console.log("Hello, " + name + "！欢迎来到JS练习页面！");
}
// 调用函数
greetUser("前端学习者");

// 2. 带返回值的函数（计算两数之和）
function addNumbers(a, b) {
    return a + b;
}
// 调用并接收返回值
let sum = addNumbers(15, 27);
console.log("15 + 27 =", sum);

// 3. 箭头函数（ES6 简化写法）
const multiply = (x, y) => x * y;
// 调用箭头函数
let product = multiply(6, 8);
console.log("6 * 8 =", product);

// 4. 高阶函数（以数组遍历为例）
let numbers = [1, 2, 3, 4, 5];
console.log("--- 数组每个元素乘2（高阶函数map） ---");
let doubleNumbers = numbers.map(num => num * 2);
console.log("原数组：", numbers);
console.log("处理后：", doubleNumbers);
