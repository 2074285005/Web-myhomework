    <script>
        // ====================== 考点1：变量定义 & 变量类型 ======================
        // 1.1 变量定义（let/const/var三种方式，推荐前两种）
        let studentName = "新闻专业-小明";  // let：可修改变量
        const studentId = 202500101;        // const：不可修改常量
        var isWebStudent = true;            // var：旧版定义方式

        // 1.2 不同变量类型练习（覆盖JS基础数据类型）
        let strType = "字符串类型";         // 字符串 String
        let numType = 98.5;                 // 数字 Number
        let boolType = false;               // 布尔 Boolean
        let nullType = null;                // 空值 Null
        let undefinedType;                  // 未定义 Undefined
        let arrType = ["HTML","CSS","JS"];  // 引用类型-数组 Array
        let objType = {name:"mgl",grade:"大二"}; // 引用类型-对象 Object

        // 控制台打印变量类型（打开F12就能看到结果）
        console.log("===== 变量类型输出 =====");
        console.log("字符串类型：", typeof strType);
        console.log("数字类型：", typeof numType);
        console.log("布尔类型：", typeof boolType);
        console.log("空值类型：", typeof nullType);
        console.log("未定义类型：", typeof undefinedType);
        console.log("数组类型：", typeof arrType);
        console.log("对象类型：", typeof objType);


        // ====================== 考点2：表达式练习 ======================
        // 2.1 算术表达式（加减乘除取余）
        let a = 15, b = 4;
        let add = a + b;    // 加法
        let sub = a - b;    // 减法
        let mul = a * b;    // 乘法
        let div = a / b;    // 除法
        let mod = a % b;    // 取余

        // 2.2 关系表达式（比较大小，返回true/false）
        let compare1 = a > b;
        let compare2 = a === b;

        // 2.3 逻辑表达式（与/或/非）
        let logic1 = (a > 10) && (b < 5);  // 逻辑与
        let logic2 = (a > 20) || (b > 2);  // 逻辑或

        // 打印表达式结果
        console.log("\n===== 表达式结果输出 =====");
        console.log("算术表达式-加减乘除取余：", add, sub, mul, div, mod);
        console.log("关系表达式-比较结果：", compare1, compare2);
        console.log("逻辑表达式-逻辑运算：", logic1, logic2);


        // ====================== 考点3：流程控制练习 ======================
        // 3.1 条件判断 if-else
        console.log("\n===== 流程控制-条件判断 =====");
        if (numType >= 90) {
            console.log("成绩评级：优秀");
        } else if (numType >= 60) {
            console.log("成绩评级：及格");
        } else {
            console.log("成绩评级：不及格");
        }

        // 3.2 循环控制 for循环
        console.log("\n===== 流程控制-for循环 =====");
        for (let i = 0; i < arrType.length; i++) {
            console.log(`第${i+1}个学习科目：${arrType[i]}`);
        }

        // 3.3 循环控制 while循环
        console.log("\n===== 流程控制-while循环 =====");
        let count = 0;
        while (count < 3) {
            console.log("循环执行次数：", count+1);
            count++;
        }


        // ====================== 考点4：函数练习 ======================
        // 4.1 定义函数（带参数、返回值）
        function calculateScoreAverage(scoreList) {
            let total = 0;
            // 遍历数组计算总分
            for (let i = 0; i < scoreList.length; i++) {
                total += scoreList[i];
            }
            // 返回平均分
            return total / scoreList.length;
        }

        // 4.2 调用函数
        let myScoreList = [88, 92, 95, 85];
        let averageScore = calculateScoreAverage(myScoreList);

        // 打印函数结果
        console.log("\n===== 函数调用结果 =====");
        console.log("我的成绩列表：", myScoreList);
        console.log("计算得出平均分：", averageScore);
    </script>
